> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implementar dois finais de jogo — Explosão (falha catastrófica do reator) e Demissão Misteriosa (falha energética com telegramas burocráticos soviéticos escalando).

**Architecture:** Dois novos módulos ES6 class (`DemandSystem`, `GameOverSystem`) integrados ao game loop existente via `update()`. DOM manipulado diretamente sem framework. Telas de game over são overlays `position:fixed` adicionadas ao `index.html`.

**Tech Stack:** Vanilla JS ES6 class, CSS3 animations, HTML5. Node.js vm para testes (mesmo padrão de `tests/reactor.test.js`).

```javascript
// ===== DEMAND SYSTEM =====
// Gerencia a cota de energia exigida pelo Partido e escala avisos burocráticos.
class DemandSystem {
    constructor(simulation) {
        this.simulation = simulation;

        this.currentQuota   = 300; // MW
        this.demandEvents   = [
            { id: 'factory', text: 'Nova fábrica em Krasnogorsk',       delta: 120 },
            { id: 'mine',    text: 'Nova mina de urânio de Volgosk',     delta: 80  },
            { id: 'city',    text: 'Nova cidade satélite de Krasnovosk', delta: 150 }
        ];
        this.firedEventIds  = [];
        this.lastDemandTime = 0;

        // Escalonamento de avisos
        this.deficiencyStart = null; // sim.time quando energia caiu abaixo da cota
        this.warningStage    = 0;    // 0=ok 1=telex1 2=telex2 3=gameover
        this.blackoutCount   = 0;

        // Callbacks definidos externamente
        this.onShowTelex = null; // fn(stage, eventText|null, quota, blackoutCount)
        this.onGameOver  = null; // fn({ time, quota, energy, blackouts })
    }

    // Chamado a cada tick pelo game loop
    update(deltaTime) {
        if (!this.simulation.running)      return;
        if (!this.simulation.eventsEnabled) return;
        this._tryDemandEvent();
        this._checkDeficiency();
    }

    getQuota() { return this.currentQuota; }

    _tryDemandEvent() {
        if (this.warningStage > 0)                              return; // aviso ativo
        if (this.firedEventIds.length >= this.demandEvents.length) return; // todos disparados
        if (this.simulation.time - this.lastDemandTime < 180000)   return; // cooldown 3 min
        if (Math.random() > 0.003)                              return; // ~1 a cada 5 min

        var next = this.demandEvents.find(function(e) {
            return !this.firedEventIds.includes(e.id);
        }, this);
        if (!next) return;

        this.firedEventIds.push(next.id);
        this.currentQuota  += next.delta;
        this.lastDemandTime = this.simulation.time;
        this.blackoutCount++;

        if (this.onShowTelex) {
            this.onShowTelex(0, next.text, this.currentQuota, this.blackoutCount);
        }
    }

    _checkDeficiency() {
        var energy = this.simulation.energyGeneration;
        var now    = this.simulation.time;

        if (energy >= this.currentQuota) {
            if (this.warningStage < 2) {
                this.deficiencyStart = null;
                this.warningStage    = 0;
            }
            return;
        }

        if (this.deficiencyStart === null) {
            this.deficiencyStart = now;
        }

        var elapsed = now - this.deficiencyStart;

        if (this.warningStage === 0 && elapsed >= 30000) {
            this.warningStage = 1;
            if (this.onShowTelex) {
                this.onShowTelex(1, null, this.currentQuota, this.blackoutCount);
            }
        } else if (this.warningStage === 1 && elapsed >= 150000) {
            this.warningStage = 2;
            this.blackoutCount++;
            if (this.onShowTelex) {
                this.onShowTelex(2, null, this.currentQuota, this.blackoutCount);
            }
        } else if (this.warningStage === 2 && elapsed >= 240000) {
            this.warningStage = 3;
            if (this.onGameOver) {
                this.onGameOver({
                    time:     this.simulation.time,
                    quota:    this.currentQuota,
                    energy:   Math.round(energy),
                    blackouts: this.blackoutCount
                });
            }
        }
    }
}

if (typeof module !== 'undefined' && module.exports) {
    module.exports = DemandSystem;
}
```

```javascript
// ===== GAME OVER SYSTEM =====
// Detecta condições de game over e exibe as telas de fim de jogo.
class GameOverSystem {
    constructor(simulation, saveSystem) {
        this.simulation  = simulation;
        this.saveSystem  = saveSystem;
        this.triggered   = false;
    }

    // Chamado a cada tick pelo game loop
    update() {
        if (this.triggered)              return;
        if (!this.simulation.running)    return;

        var s = this.simulation;
        var t = REACTOR_CONFIG.alarmThresholds;
        var cause = null;

        if (s.coreTemperature >= t.temp.critical) {
            cause = {
                ru:    'Суперперегрев активной зоны',
                pt:    'Superaquecimento do núcleo',
                param: 'ТЕМПЕРАТУРА',
                value: s.coreTemperature.toFixed(0) + '°C'
            };
        } else if (s.pressure >= t.pressure.critical) {
            cause = {
                ru:    'Разрыв первичного контура',
                pt:    'Ruptura do circuito primário',
                param: 'ДАВЛЕНИЕ',
                value: s.pressure.toFixed(1) + ' МПа'
            };
        } else if (s.radiationLevel >= t.radiation.critical) {
            cause = {
                ru:    'Радиационное заражение',
                pt:    'Contaminação por radiação',
                param: 'РАДИАЦИЯ',
                value: s.radiationLevel.toFixed(1) + ' мЗв/ч'
            };
        }

        if (cause) {
            this.triggered = true;
            this.simulation.stop();
            this.saveSystem.clear();
            this._showExplosionScreen({
                cause:           cause,
                coreTemperature: s.coreTemperature,
                pressure:        s.pressure,
                radiationLevel:  s.radiationLevel,
                time:            s.time
            });
        }
    }

    // Chamado pelo DemandSystem quando os 3 telegramas se esgotam
    triggerDismissal(stats) {
        if (this.triggered) return;
        this.triggered = true;
        this.saveSystem.clear();
        this._showDismissalScreen(stats);
    }

    // Exibe popup de telegrama e pausa o jogo
    showTelex(stage, eventText, quota, blackoutCount) {
        var self = this;
        this.simulation.running = false;

        var modal    = document.getElementById('telex-modal');
        var headerEl = document.getElementById('telex-header');
        var refEl    = document.getElementById('telex-ref');
        var bodyEl   = document.getElementById('telex-body');
        var quotaEl  = document.getElementById('telex-quota');
        var btnEl    = document.getElementById('btn-telex-ok');
        if (!modal) { this.simulation.running = true; return; }

        var refNum = 1000 + Math.floor(Math.random() * 8000);

        if (stage === 0) {
            headerEl.textContent = 'МИНГОСЭНЕРГО СССР — ПОВЫШЕНИЕ НОРМЫ';
            refEl.textContent    = 'Ref: ' + refNum + '-Б | KJNP-1';
            bodyEl.innerHTML     = 'Em virtude de <strong>' + eventText + '</strong>, ' +
                                   'a demanda energética de Krasnostan foi atualizada.<br><br>' +
                                   'Confiamos em sua dedicação ao Partido.';
            quotaEl.textContent  = 'Nova cota obrigatória: ' + quota + ' MW';
            quotaEl.style.color  = '#3a2a1a';
        } else if (stage === 1) {
            headerEl.textContent = 'МИНГОСЭНЕРГО СССР — ВНИМАНИЕ';
            refEl.textContent    = 'Ref: ' + refNum + '-В | KJNP-1';
            bodyEl.innerHTML     = 'Tovarisch operador,<br><br>' +
                                   'Apagões registrados em <strong>' + blackoutCount + '</strong> regiões. ' +
                                   'A produção atual é insatisfatória.<br><br>' +
                                   'O Ministério aguarda sua resposta imediata.';
            quotaEl.textContent  = 'Cota: ' + quota + ' MW — PRAZO FINAL SE APROXIMA';
            quotaEl.style.color  = '#cc6600';
        } else if (stage === 2) {
            headerEl.textContent = 'МИНГОСЭНЕРГО СССР — ПОСЛЕДНЕЕ ПРЕДУПРЕЖДЕНИЕ';
            refEl.textContent    = 'Ref: СРОЧНО | KJNP-1';
            bodyEl.innerHTML     = 'Tovarisch <strong>████████</strong>,<br><br>' +
                                   'Sua contribuição foi… notada pelo Partido.<br>' +
                                   'Uma visita de cortesia foi organizada.<br><br>' +
                                   'Traga agasalho.';
            quotaEl.textContent  = 'ÚLTIMA NOTIFICAÇÃO';
            quotaEl.style.color  = '#cc2222';
        }

        modal.style.display = 'flex';
        btnEl.onclick = function() {
            modal.style.display = 'none';
            if (!self.triggered) self.simulation.running = true;
        };
    }

    _formatTime(ms) {
        var total = Math.floor(ms / 1000);
        var min   = Math.floor(total / 60);
        var sec   = total % 60;
        return (min < 10 ? '0' : '') + min + ':' + (sec < 10 ? '0' : '') + sec;
    }

    _showExplosionScreen(data) {
        var screen = document.getElementById('gameover-explosion');
        if (!screen) return;
        document.getElementById('exp-cause-ru').textContent  = data.cause.ru;
        document.getElementById('exp-cause-pt').textContent  = '(' + data.cause.pt + ')';
        document.getElementById('exp-param').textContent     = data.cause.param + ': ' + data.cause.value;
        document.getElementById('exp-temp').textContent      = data.coreTemperature.toFixed(0) + '°C';
        document.getElementById('exp-pressure').textContent  = data.pressure.toFixed(1) + ' МПа';
        document.getElementById('exp-radiation').textContent = data.radiationLevel.toFixed(1) + ' мЗв/ч';
        document.getElementById('exp-time').textContent      = this._formatTime(data.time);
        screen.style.display = 'flex';
    }

    _showDismissalScreen(stats) {
        var screen = document.getElementById('gameover-dismissal');
        if (!screen) return;
        document.getElementById('dis-time').textContent     = this._formatTime(stats.time);
        document.getElementById('dis-blackouts').textContent = String(stats.blackouts);
        document.getElementById('dis-quota').textContent    = stats.quota + ' MW';
        document.getElementById('dis-deficit').textContent  = (stats.quota - stats.energy) + ' MW';
        screen.style.display = 'flex';
    }
}

if (typeof module !== 'undefined' && module.exports) {
    module.exports = GameOverSystem;
}
```
